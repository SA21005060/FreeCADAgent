```python
import FreeCAD
import FreeCADGui
import Part

# ==========================================
# 1. 定义参数 (解决 name 'fuselage_length' is not defined 错误)
# ==========================================
fuselage_length = 10.0   # 机身长度
fuselage_radius = 1.0   # 机身半径
vtail_height = 3.0      # 垂直尾翼高度
vtail_chord = 1.5       # 垂直尾翼弦长

# ==========================================
# 2. 创建文档和部件
# ==========================================

# 创建新文档
App.newDocument("Airplane")
App.setActiveDocument("Airplane")
App.ActiveDocument = App.getDocument("Airplane")
Gui.ActiveDocument = Gui.getDocument("Airplane")

# --- 创建机身 (Fuselage) ---
# 使用圆柱体，沿X轴放置
fuselage = App.ActiveDocument.addObject("Part::Cylinder", "Fuselage")
fuselage.Radius = fuselage_radius
fuselage.Height = fuselage_length
# 旋转90度使其沿X轴躺平
fuselage.Placement = FreeCAD.Placement(
    FreeCAD.Vector(0, 0, 0), 
    FreeCAD.Rotation(FreeCAD.Vector(1, 0, 0), 90)
)

# --- 创建主翼 (Main Wing) ---
wing = App.ActiveDocument.addObject("Part::Box", "MainWing")
wing.Length = 2.0     # 翼弦 (X方向)
wing.Height = 0.2    # 翼厚 (Y方向)
wing.Width = 12.0    # 翼展 (Z方向)
# 放置在机身中心
wing.Placement = FreeCAD.Placement(
    FreeCAD.Vector(0, 0, 0), 
    FreeCAD.Rotation(0, 0, 0)
)

# --- 创建垂直尾翼 (Vertical Tail) ---
# 这里使用了您之前报错的代码逻辑
vtail = App.ActiveDocument.addObject("Part::Box", "Vtail")
vtail.Length = vtail_chord  # 弦长 (X方向)
vtail.Height = 0.2          # 厚度 (Y方向)
vtail.Width = vtail_height  # 高度 (Z方向 - 垂直)

# 定位逻辑 (修正后的版本，现在变量已定义)
# 位置 X: 机身尾部 (-fuselage_length/2) 往回一点
# 位置 Z: 机身半径 + 尾翼高度的一半
vtail.Placement = FreeCAD.Placement(
    FreeCAD.Vector(-fuselage_length/2 + 1, 0, vtail_height/2 + 0.5), 
    FreeCAD.Rotation(FreeCAD.Vector(1, 0, 0), 0)
)

# 重新计算以更新模型
App.ActiveDocument.recompute()

# ==========================================
# 3. 视图设置 (等轴测视图并适应模型)
# ==========================================
Gui.ActiveDocument.ActiveView.viewIsometric()
Gui.ActiveDocument.ActiveView.fitAll()
```

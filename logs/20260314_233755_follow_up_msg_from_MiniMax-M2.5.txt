```python
def create_cilinder(radius, height, position, rotation=(0,0,0), color=color_skin):
    # Cylinder: height is along Z. 
    # Position is the center of the cylinder.
    # We need to move it so the base is at z=0 relative to placement?
    # Actually, placing by center is easier.
    cyl = App.ActiveDocument.addObject("Part::Cylinder", "Cylinder")
    cyl.Radius = radius
    cyl.Height = height
    cyl.Placement = App.Placement(
        App.Vector(*position), 
        App.Rotation(*rotation)
    )
    cyl.ViewObject.ShapeColor = color
    return cyl

def create_box(dims, position, rotation=(0,0,0), color=color_skin):
    box = App.ActiveDocument.addObject("Part::Box", "Box")
    box.Width = dims[0]
    box.Height = dims[1]
    box.Length = dims[2]
    # Box placement is corner? No, in FreeCAD Part::Box placement is center if created via GUI?
    # No, standard Box is placed at origin. We need to translate.
    # Let's use a transformation.
    box.Placement = App.Placement(
        App.Vector(*position),
        App.Rotation(*rotation)
    )
    box.ViewObject.ShapeColor = color
    return box
```

```python
import FreeCAD
import Part
import math

def createShape(objType, params, pos, rot, color):
    if objType == 'Sphere':
        obj = FreeCAD.ActiveDocument.addObject("Part::Sphere", "Sphere")
        obj.Radius = params['radius']
    elif objType == 'Cylinder':
        obj = FreeCAD.ActiveDocument.addObject("Part::Cylinder", "Cylinder")
        obj.Radius = params['radius']
        obj.Height = params['height']
    elif objType == 'Box':
        obj = FreeCAD.ActiveDocument.addObject("Part::Box", "Box")
        obj.Length = params['length']
        obj.Width = params['width']
        obj.Height = params['height']
    
    obj.Placement = FreeCAD.Placement(FreeCAD.Vector(*pos), FreeCAD.Rotation(*rot))
    obj.ViewObject.ShapeColor = color
    return obj

# Clear document
for obj in FreeCAD.ActiveDocument.Objects:
    FreeCAD.ActiveDocument.removeObject(obj.Name)

# Definitions
# Position Z=0 is bottom of feet
# Head Z ~ 80
# Body Z ~ 55
# Legs Z ~ 20

skin_color = (1.0, 0.85, 0.73)  # Light skin tone
shirt_color = (0.2, 0.6, 0.8)   # Blue shirt
pants_color = (0.3, 0.3, 0.6)   # Dark blue pants
shoes_color = (0.1, 0.1, 0.1)   # Black
hair_color = (0.3, 0.2, 0.1)    # Brown

# --- Legs (Pants) ---
# Left Leg
createShape('Cylinder', {'radius': 6, 'height': 40}, (-6, 0, 20), (0,0,0), pants_color)
# Right Leg
createShape('Cylinder', {'radius': 6, 'height': 40}, (6, 0, 20), (0,0,0), pants_color)

# --- Feet (Shoes) ---
createShape('Box', {'length': 14, 'width': 10, 'height': 8}, (-6, 0, 4), (0,0,0), shoes_color)
createShape('Box', {'length': 14, 'width': 10, 'height': 8}, (6, 0, 4), (0,0,0), shoes_color)

# --- Torso (Shirt) ---
# A rounded box or just a box. Let's use a box with slightly wider dimensions.
createShape('Box', {'length': 22, 'width': 12, 'height': 40}, (0, 0, 50), (0,0,0), shirt_color)

# --- Arms (Sleeves + Hands) ---
# Left Arm
createShape('Cylinder', {'radius': 5, 'height': 20}, (-16, 0, 50), (0,0,0), shirt_color) # Sleeve
createShape('Cylinder', {'radius': 3, 'height': 15}, (-16, 0, 32), (0,0,0), skin_color) # Hand

# Right Arm
createShape('Cylinder', {'radius': 5, 'height': 20}, (16, 0, 50), (0,0,0), shirt_color) # Sleeve
createShape('Cylinder', {'radius': 3, 'height': 15}, (16, 0, 32), (0,0,0), skin_color) # Hand

# --- Head ---
# Sphere
createShape('Sphere', {'radius': 14}, (0, 0, 84), (0,0,0), skin_color)

# --- Hair ---
# A slightly smaller sphere shifted up and slightly forward
createShape('Sphere', {'radius': 14}, (0, 1, 88), (0,0,0), hair_color)
# Bangs
createShape('Sphere', {'radius': 12}, (0, 8, 86), (0,0,0), hair_color)

# --- Face (Eyes) ---
# Simple white spheres
createShape('Sphere', {'radius': 2}, (-5, 10, 86), (0,0,0), (1,1,1))
createShape('Sphere', {'radius': 2}, (5, 10, 86), (0,0,0), (1,1,1))
# Pupils (Black)
createShape('Sphere', {'radius': 1}, (-5, 11, 87), (0,0,0), (0,0,0))
createShape('Sphere', {'radius': 1}, (5, 11, 87), (0,0,0), (0,0,0))

# --- View Settings ---
# Switch to isometric view
FreeCADGui.activeDocument().activeView().viewIsometric()
# Fit everything in view
FreeCADGui.activeDocument().activeView().fitAll()
```
